openssl req -x509 -newkey rsa:4096 -keyout domain.key -out domain.crt -days 365 -subj '/CN=rockies.local'

openssl rsa -in domain.key -out domain.unencrypted.key -passin pass:simple

docker run -d \
  --restart=always \
  --name registry \
  -v `pwd`/certs:/certs \
  -e REGISTRY_HTTP_ADDR=0.0.0.0:443 \
  -e REGISTRY_HTTP_TLS_CERTIFICATE=/certs/domain.crt \
  -e REGISTRY_HTTP_TLS_KEY=/certs/domain.unencrypted.key \
  -p 443:443 \
  registry
